for j in {1..5}; do 
dir="/home/lezh/dsmwpred/zly"
echo "#"'!'"/bin/bash
#SBATCH --mem 2G
#SBATCH -t 10:0:0
#SBATCH -c 1
#SBATCH -A dsmwpred
#SBATCH --constraint \"s05\"


source /home/lezh/miniconda3/etc/profile.d/conda.sh

${dir}/software/plink --bfile ${dir}/data_1Wan --linear hide-covar --covar ${dir}/covar_1Wan_PC_10_withoutLabel.covars --pheno ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Trait_19.pheno  --mpheno $j --allow-no-sex --memory 2048 --out ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Result_Trait19/Result_Plink_Trait19_P$j

" > ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_13_to_24/Result_Plink/Result_Plink_Trait19_P$j

# I am doing blabla
cd ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_13_to_24/Result_Plink

sbatch Result_Plink_Trait19_P$j
done

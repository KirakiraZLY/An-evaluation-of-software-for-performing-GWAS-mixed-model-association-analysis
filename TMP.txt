" > ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_13_to_24/Result_Plink/Result_Plink_Trait13_P$j

# I am doing blabla
cd ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_13_to_24/Result_Plink

sbatch Result_Plink_Trait13_P$j
done

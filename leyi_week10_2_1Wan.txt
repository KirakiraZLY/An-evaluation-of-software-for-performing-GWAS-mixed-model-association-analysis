for j in {1..5}; do 
dir="/home/lezh/dsmwpred/zly"
echo "#"'!'"/bin/bash
#SBATCH --mem 2G
#SBATCH -t 10:0:0
#SBATCH -c 1
#SBATCH -A dsmwpred
#SBATCH --constraint \"s05\"


source /home/lezh/miniconda3/etc/profile.d/conda.sh

${dir}/software/plink --bfile ${dir}/data_qc --linear hide-covar --covar ${dir}/covar_PC_withoutLabel.covars --pheno ${dir}/type_1_error/Multi_Traits/Trait_1_to_12/Trait_1.pheno  --mpheno $j --allow-no-sex --memory 2048 --out ${dir}/type_1_error/Multi_Traits/Trait_1_to_12/Result_Trait1/Result_Plink_Trait1_P$j

" > ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_1_to_12/Result_Plink/Result_Plink_Trait1_P$j

# I am doing blabla
cd ${dir}/scripts/type_1_error_new/Multi_Traits/Trait_1_to_12/Result_Plink

sbatch Result_Plink_Trait1_P$j
done

" > ${dir}/scripts/type_1_error_new/Multi_Traits/META_ANALYSIS/T1/Result_LDAK_White_P$j

# I am doing blabla
cd ${dir}/scripts/type_1_error_new/Multi_Traits/META_ANALYSIS/T1
sbatch Result_LDAK_White_P$j
done

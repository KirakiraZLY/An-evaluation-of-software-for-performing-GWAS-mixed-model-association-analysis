  --bed ${dir}/MAMA/Bolt_Height/data_Black \

  --covarCol PC{1:20} \

  --threads 4 \
  --out ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Result_Trait$i/Result_Trait$i_NEW_regenie_s1

  --step 2 \
  --bgen ${dir}/MAMA/Bolt_Height/data_Black.bgen \
  --covarFile ${dir}/covar_1Wan_PC.covars \
  --phenoFile ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Trait_$i_label.pheno \
  --covarCol PC{1:20} \
  --bsize 1000 \
  --threads 4 \
  --qt \
  --pThresh 0.01 \
  --pred ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Result_Trait$i/Result_Trait$i_NEW_regenie_s1_pred.list \
  --out ${dir}/type_1_error/Multi_Traits/Trait_13_to_24/Result_Trait$i/Result_Trait$i_NEW_regenie_s2
